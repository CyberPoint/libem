00515          * 1. construct an n X k "posterior" matrix of weighted density values p_nk = P(n|k)*P(k)/Q_n

00533         {
00534                 retcode = compute_norm_constants(posteriors,norm_constants);

00548

00559         {
00560                 retcode = compute_weighted_means(X,posteriors,norm_constants,weighted_means);
00561
