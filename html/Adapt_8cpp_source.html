00482                 Matrix &mu_matrix, Matrix &Pks,

00486         int num_dimensions = mu_matrix.colCount();              // number of data dimensions
00487 
00488         int retcode = 1;
00489         /*
00490          * 1. construct an n X k "posterior" matrix of weighted density values p_nk = P(n|k)*P(k)/Q_n
00491          * for each data point, where Q_n is the normalization factor given by the sum of P(n|k)*P(k) over all k.
00492         */

00504          *  data point, under the cluster.
00505          */

00507         if (retcode != 0)
00508         {
00509                 retcode = compute_norm_constants(posteriors,norm_constants);
00510         }
00511

00520                 for (int i = 0; i < num_clusters; i++)

00523 
00524                 }
00525         }

00533         if (retcode != 0)
00534         {
00535                 retcode = compute_weighted_means(X,posteriors,norm_constants,weighted_means);
00536 
00537         }

00549                 for (int i = 0; i < num_clusters; i++)

00555 
00556         }
00557

00559          *  6. now compute the new cluster weights W_i = Y [(a_i * v_i/N) + (1-a_i)*w_i
